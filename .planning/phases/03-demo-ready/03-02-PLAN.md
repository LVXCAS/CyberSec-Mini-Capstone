Create `scripts/reset.sh`:
```bash
#!/usr/bin/env bash
set -euo pipefail
cd "$(dirname "$0")/.."

echo "=== Tearing down all containers and volumes ==="
docker compose down -v --remove-orphans 2>/dev/null || true

echo "=== Rebuilding from scratch ==="
docker compose up --build -d

echo "=== Waiting for services to be healthy ==="
SERVICES=("orchestrator")  # Add other services with health checks here
for svc in "${SERVICES[@]}"; do
  echo -n "Waiting for $svc..."
  timeout 120 bash -c "
    until docker compose ps $svc --format json 2>/dev/null | grep -q healthy; do
      sleep 2
      echo -n '.'
    done
  " || { echo " TIMEOUT"; exit 1; }
  echo " healthy"
done

echo "=== Reset complete ==="
```

If the orchestrator container doesn't have a HEALTHCHECK in docker-compose.yml, add a simpler wait: poll `curl -sf http://localhost:8000/health` in a loop with 120s timeout instead of docker health status.

Create `scripts/validate-reset.sh`:
```bash
#!/usr/bin/env bash
set -euo pipefail
cd "$(dirname "$0")/.."

RUNS=5
echo "=== Validating $RUNS consecutive reset+start cycles ==="

for i in $(seq 1 $RUNS); do
  echo ""
  echo "--- Run $i/$RUNS ---"
  ./scripts/reset.sh

  # Verify health
  if ! curl -sf http://localhost:8000/health > /dev/null 2>&1; then
    echo "FAIL: Run $i — orchestrator not healthy after reset"
    exit 1
  fi

  # Verify no stale game data (score_events should be empty after fresh start)
  SCORE_COUNT=$(curl -sf http://localhost:8000/game/status 2>/dev/null | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('scores',{}).get('red',0) + d.get('scores',{}).get('blue',0))" 2>/dev/null || echo "0")
  if [ "$SCORE_COUNT" != "0" ]; then
    echo "FAIL: Run $i — stale score data detected ($SCORE_COUNT points)"
    exit 1
  fi

  echo "Run $i: PASS"
done

echo ""
echo "=== All $RUNS runs passed — no stale data ==="
```

Make executable: `chmod +x scripts/validate-reset.sh`
